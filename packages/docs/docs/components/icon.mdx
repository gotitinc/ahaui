import { useState, useMemo } from "react";
import classNames from "classnames";
import { Icon, Form, Overlay, Tooltip } from "@ahaui/react";
import CodeView from "@site/src/components/CodeView";
import colorIcon from "@site/examples/Icon/color";
import sizeIcon from "@site/examples/Icon/size";

import StbIconGrid1 from "@site/assets/structure/stb_icon-grid.jpg";
import StbIconGrid2 from "@site/assets/structure/stb_icon-grid@2x.jpg";
import StbIconShapes1 from "@site/assets/structure/stb_icon-shapes.jpg";
import StbIconShapes2 from "@site/assets/structure/stb_icon-shapes@2x.jpg";


export const AllIcons = () => {
  {/* TODO: Use `icons` from @ahaui/react when local aha package is */}
  const icons = useMemo(() => ({}), []);
  const [iconDisplay, setIconDisplay] = useState(null);
  const [searchText, setSearchText] = useState("");
  const handleSearch = (e) => {
    const value = e.target.value;
    setIconDisplay(null);
    setSearchText(value.toLowerCase());
  };
  return (
    <>
      <Form.Group controlId="search-icon" sizeControl="small">
        <Form.Input
          value={searchText}
          onChange={handleSearch}
          placeholder="Search icon ..."
        />
      </Form.Group>
      {iconDisplay && (
        <CodeView
          horizontal
          show
          codeText={`<Icon size="medium" name="${iconDisplay}"/>`}
        />
      )}
      <div className="u-positionRelative u-paddingSmall u-marginBottomExtraSmall u-border u-borderSmall u-borderUltraLight u-roundedMedium">
        {Object.keys(icons)
          .filter((item) => item.includes(searchText))
          .map((item) => (
            <div
              onClick={() => setIconDisplay(item)}
              key={item}
              className={classNames(
                "u-marginRightSmall u-marginBottomSmall u-cursorPointer u-inlineBlock",
                iconDisplay === item && "u-textPrimary"
              )}
            >
              <Overlay.Trigger
                placement="bottom"
                delay={{ show: 500, hide: 1000 }}
                hoverOverlay
                overlay={(props) => (
                  <Tooltip id="tooltip-icon-search" {...props}>
                    {item}
                  </Tooltip>
                )}
              >
                <Icon name={item} size="medium" />
              </Overlay.Trigger>
            </div>
          ))}
      </div>
    </>
  );
};

## Introduction

Icons are used to visualize the common actions, provide visual context and enhance usability. They should be simple, yet bold enough to grab users' attention.

### Icon grid

Regular icons are designed on an 18px grid and exported on a 24px artboard.
